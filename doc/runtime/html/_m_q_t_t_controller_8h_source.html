230inline bool CMQTTController::isDeviceTopic(const char *pszTopic) {
231 bool bIsDeviceTopic = false;
232 if(pszTopic) {
233 size_t nPublishTopicLength = Config.PublishTopicPrefix.length();
234 if(strlen(pszTopic) > nPublishTopicLength && pszTopic[nPublishTopicLength] == '/') {
235 String strTopicPrefix(pszTopic,nPublishTopicLength);
236 bIsDeviceTopic = LSC::stricmp(strTopicPrefix.c_str(),Config.PublishTopicPrefix.c_str()) == 0;
237 }
238 }
239 return(bIsDeviceTopic);
240}
241
242inline bool CMQTTController::isDeviceCommandTopic(const char *pszTopic) {
243 bool bIsCommandTopic = false;
244 if(isDeviceTopic(pszTopic)) {
245 size_t nPublishTopicLength = Config.PublishTopicPrefix.length();
246 bIsCommandTopic = LSC::stricmp(&pszTopic[nPublishTopicLength],"/cmd") == 0 ||
247 strncmp(&pszTopic[nPublishTopicLength],"/cmd/",5) == 0;
248 }
249 return(bIsCommandTopic);
250}
