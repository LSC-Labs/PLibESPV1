37// The COMPILE-TIME SETTING SAFEBASE64 is really important.
38// You need to decide if PARANOIA is more important to you than speed.
39//
40// SAFEBASE64: Remove this def to NOT check the validity of base64 ascii strings
41// before unbase64'ing that string. If you don't #define SAFEBASE64,
42// then the program assumes that all characters in the string sent to unbase64()
43// are in the base64 alphabet. As such if a character is NOT in the base64 alphabet
44// your data will be wrong (it will be turned to 0 (as if it were just a base64 'A')).
45// Removing this test greatly speeds up unbase64'ing (by about 3-4x).
46#define SAFEBASE64
47#define isMultipleOf(a,x) (!((a)%x))
48
49// Converts any binary data to base64 characters.
50// Length of the resultant string is stored in flen
51// (you must pass pointer flen).
52char* base64( const void* binaryData, int len, int *flen );
53
54// Convert your base64 string haJIh/+ back to binary data.
55// len is the string length and should NOT include the null terminator.
56// Final size will be stored in flen
57// (you must pass pointer flen).
58unsigned char* unbase64( const char* ascii, int len, int *flen );

60// Checks the integrity of a base64 string to make sure it is
61// made up of only characters in the base64 alphabet (array b64)
62#define isbase64ValidChr( ch ) ( ('0' <= ch && ch <= '9') || \
63 ('A' <= ch && ch <= 'Z') || ('a' <= ch && ch <= 'z') || \
64 ch=='+' || ch=='/' ) // other 2 valid chars, + ending chrs
65// = is NOT considered a valid base64 chr, it's only valid at the end for padding
66
67// Tells you if a string is valid base64, which means it's length is
68// a multiple of 4, and it contains only valid base64 chrs.
69int base64integrity( const char *ascii, int len );
